1 Khai báo hàm & return Cơ bản

function sum(a, b) {        // a, b: THAM SỐ (parameter)
  return a + b;             // return: trả giá trị về nơi gọi & THOÁT hàm
}

sum(5, 6);                    // gọi hàm với ĐỐI SỐ (argument) 5, 6 → 11
let kq = sum(2, 3) * 10;      // dùng kết quả như một giá trị → 50

Hàm không có return (hoặc return không kèm giá trị) sẽ trả về undefined. Sau khi return chạy, mọi lệnh phía dưới trong hàm bị bỏ qua — tận dụng để "thoát sớm" khi dữ liệu không hợp lệ.

2 Tham số & đối số — JS "dễ dãi" thế nào Cơ bản

JavaScript không kiểm tra kiểu, không kiểm tra số lượng đối số. Thiếu đối số → tham số nhận undefined.

function sum(a, b) { return a + b; }

sum(5, 6);      // 11  — đủ đối số
sum(5);         // NaN — b = undefined → 5 + undefined = NaN
sum(5, 6, 99);  // 11  — đối số thừa bị lờ đi
sum("5", 6);    // "56" — không ai chặn bạn truyền chuỗi!

3 Tham số mặc định ES6

function add(a, b = 5) {      // b không được truyền → tự lấy 5
  return a + b;
}
add(5);      // 10
add(5, 7);   // 12 — có truyền thì dùng giá trị truyền

// ứng dụng thật: lời chào có tên mặc định
function chao(ten = "bạn") {
  return `Xin chào ${ten}!`;
}

4 Tham số rest ... ES6

Gom số lượng đối số bất kỳ thành một mảng — hết cảnh "không biết người ta truyền mấy số".

function sum(...nums) {        // nums là MẢNG các đối số
  let s = 0;
  for (let x of nums) s += x;
  return s;
}
sum(4, 5, 2);        // 11
sum(1, 2, 3, 4, 5);  // 15 — bao nhiêu cũng nhận
Thử ngay — cộng bao nhiêu số tuỳ bạn
// nhập dãy số cách nhau dấu phẩy

5 Function expression Nâng cao

Hàm (thường không tên) gán vào biến — vì trong JS, hàm cũng là một giá trị như số hay chuỗi.

// function DECLARATION — có tên, được "hoisting" (gọi trước khai báo vẫn chạy)
function tinhTong(a, b) { return a + b; }

// function EXPRESSION — gán hàm vô danh vào biến, KHÔNG hoisting
let x = function (a, b) { return a + b; };
x(10, 20);   // 30 — gọi qua tên biến

Hoisting khác nhau: declaration được "kéo lên đầu" nên gọi trước khi khai báo vẫn chạy; expression thì không — gọi trước dòng gán sẽ lỗi. Đây là khác biệt thực dụng nhất giữa hai cách viết.

6 Arrow function => ES6

Cú pháp ngắn gọn của function expression — chuẩn viết callback hiện đại.

// 4 nấc rút gọn — cùng một hàm cộng:
let add = function (a, b) { return a + b; };  // ① expression đầy đủ
let add = (a, b) => { return a + b; };         // ② thay function bằng =>
let add = (a, b) => a + b;                     // ③ thân 1 biểu thức: bỏ { } và return
let binhPhuong = x => x * x;                   // ④ 1 tham số: bỏ luôn ( )

this của arrow khác hàm thường! Arrow không có this riêng — nó "mượn" this của phạm vi bao ngoài. Khi xử lý sự kiện DOM mà cần this là phần tử nhận sự kiện thì phải dùng function () {} thường — gặp lại bẫy này ở bài 3.8.

7 Arrow + map / filter / sort Nâng cao

Lý do arrow phổ biến: truyền hàm ngắn gọn vào các phương thức mảng.

let a = [5, 8, 5, 9, -9, 4];

a.map(v => v + 2);        // [7, 10, 7, 11, -7, 6]   — tăng 2 mỗi phần tử
a.filter(v => v % 2);     // [5, 5, 9, -9]           — lọc số lẻ (1 là truthy)
a.sort((t1, t2) => t2 - t1); // [9, 8, 5, 5, 4, -9]  — sắp GIẢM dần
Thử ngay — bộ ba map · filter · sort trên [5, 8, 5, 9, -9, 4]
// mỗi nút một phép biến đổi — mảng gốc không bị map/filter làm hỏng

8 Hàm thuần & tái sử dụng Chuẩn nghề

Hàm thuần (pure function): cùng đầu vào → cùng đầu ra, không tác dụng phụ — dễ test, dễ tái sử dụng.

// ✗ KHÔNG thuần: phụ thuộc & sửa biến bên ngoài, khó tái sử dụng
let tong = 0;
function congDon(x) { tong += x; }

// ✓ THUẦN: chỉ nhìn tham số, chỉ trả kết quả
const tinhBmi = (kg, m) => kg / (m * m);
// dùng lại được ở mọi nơi: form BMI, bảng thống kê, kiểm thử…

Rubric thực hành "viết hàm xử lý cần thiết": tách phần tính toán (hàm thuần — nhận tham số, trả kết quả) khỏi phần giao diện (đọc input, ghi DOM). Bài tập BMI ở bài 14 áp dụng đúng mẫu này.

9 Callback — hàm truyền vào hàm Nâng cao

Vì hàm là giá trị, ta truyền được hàm A vào hàm B để B "gọi lại" (call back) khi cần — nền tảng của sự kiện (3.7) và bất đồng bộ (3.12).

// map nhận CALLBACK: gọi lại hàm đó với từng phần tử
[1, 2, 3].map(x => x * 10);              // [10, 20, 30]

// addEventListener nhận CALLBACK: gọi lại khi có click
nut.addEventListener("click", () => console.log("Đã bấm!"));

// setTimeout nhận CALLBACK: gọi lại sau 2 giây
setTimeout(() => console.log("2 giây trôi qua"), 2000);
Thử ngay — callback hẹn giờ
// bấm rồi chờ — callback chạy SAU, không chặn trang

10 Mẹo & lỗi thường gặp Tổng kết

Lỗi 1. Gọi thiếu đối số → undefined → kết quả NaN. Phòng bằng tham số mặc định hoặc kiểm tra đầu hàm.

Lỗi 2. nut.addEventListener("click", sayHello()) — cặp ngoặc thừa khiến hàm chạy NGAY thay vì đợi click. Truyền tên hàm: sayHello, không phải kết quả gọi hàm.

Lỗi 3. Quên return — hàm tính đúng nhưng trả về undefined vì kết quả chỉ nằm trong console.log.

Ghi nhớ. Mặc định viết arrow cho callback; hàm có tên dùng declaration; cần this phần tử sự kiện → function thường; tách hàm thuần khỏi code giao diện.
